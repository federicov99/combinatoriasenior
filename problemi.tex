}
\newcommand{\PrintTag}[1]{%
	\subsection{#1}
	\it=0%
	\loop
		\def\listoftags{\csname Tags\the\it\endcsname}%
		\IfSubStringInString{#1}{\listoftags}
		{\PrintShort{\it}}\relax%
		\advance\it by 1
	\ifnum\it < \the\problemi
	\repeat
}
